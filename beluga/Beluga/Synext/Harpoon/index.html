Module Synext.Harpoon

Syntax of Harpoon commands.

type defer_kind = [
  1. | `subgoal
  2. | `theorem
]
type invoke_kind = [
  1. | `ih
  2. | `lemma
]
type split_kind = [
  1. | `split
  2. | `invert
  3. | `impossible
]
type level = [
  1. | `meta
  2. | `comp
]
type automation_kind = [
  1. | `auto_intros
  2. | `auto_solve_trivial
]
type automation_change = [
  1. | `on
  2. | `off
  3. | `toggle
]
type basic_command = [
  1. | `list
  2. | `defer
]
type info_kind = [
  1. | `prog
]
type command =
  1. | Rename of {
    1. rename_from : Name.t;
    2. rename_to : Name.t;
    3. level : level;
    }
  2. | ToggleAutomation of automation_kind * automation_change
  3. | Type of Comp.exp
  4. | Info of info_kind * Name.t
  5. | SelectTheorem of Name.t
  6. | Theorem of [ basic_command | `show_ihs | `show_proof | `dump_proof of string ]
  7. | Session of [ basic_command | `create | `serialize ]
  8. | Subgoal of basic_command
  9. | Undo
  10. | Redo
  11. | History
  12. | Translate of Name.t
  13. | Intros of string list option
  14. | Split of split_kind * Comp.exp
  15. | MSplit of Beluga_syntax.Location.t * Name.t
  16. | Solve of Comp.exp
  17. | Unbox of Comp.exp * Name.t * Comp.unbox_modifier option
  18. | By of Comp.exp * Name.t
  19. | Suffices of Comp.exp * Comp.suffices_typ list
  20. | Help
