External Signature Syntax

type datatype_flavour =
  1. | InductiveDatatype
  2. | StratifiedDatatype
type precedence = int
type pragma =
  1. | NamePrag of Name.t * string * string option
  2. | FixPrag of Name.t * Beluga_syntax.Fixity.t * precedence * Beluga_syntax.Associativity.t option
  3. | NotPrag
  4. | DefaultAssocPrag of Beluga_syntax.Associativity.t
  5. | OpenPrag of string list
  6. | AbbrevPrag of string list * string
type global_pragma =
  1. | NoStrengthen
  2. | Coverage of [ `Error | `Warn ]
type thm_decl =
  1. | Theorem of {
    1. location : Beluga_syntax.Location.t;
    2. name : Name.t;
    3. typ : Comp.typ;
    4. order : Comp.total_dec option;
    5. body : Comp.thm;
    }
type decl =
  1. | Typ of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. kind : LF.kind;
    }
    (*

    LF type family declaration

    *)
  2. | Const of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. typ : LF.typ;
    }
    (*

    LF type constant declaration

    *)
  3. | CompTyp of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. kind : Comp.kind;
    4. datatype_flavour : datatype_flavour;
    }
    (*

    Computation-level data type constant declaration

    *)
  4. | CompCotyp of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. kind : Comp.kind;
    }
    (*

    Computation-level codata type constant declaration

    *)
  5. | CompConst of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. typ : Comp.typ;
    }
    (*

    Computation-level type constructor declaration

    *)
  6. | CompDest of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. mctx : LF.mctx;
    4. observation_typ : Comp.typ;
    5. return_typ : Comp.typ;
    }
    (*

    Computation-level type destructor declaration

    *)
  7. | CompTypAbbrev of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. kind : Comp.kind;
    4. typ : Comp.typ;
    }
    (*

    Synonym declaration for computation-level type

    *)
  8. | Schema of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. schema : LF.schema;
    }
    (*

    Declaration of a specification for a set of contexts

    *)
  9. | Pragma of {
    1. location : Beluga_syntax.Location.t;
    2. pragma : pragma;
    }
    (*

    Compiler directive

    *)
  10. | GlobalPragma of {
    1. location : Beluga_syntax.Location.t;
    2. pragma : global_pragma;
    }
    (*

    Global directive

    *)
  11. | MRecTyp of {
    1. location : Beluga_syntax.Location.t;
    2. declarations : (decl * decl list) Support.List1.t;
    }
    (*

    Mutually-recursive LF type family declaration

    *)
  12. | Theorems of {
    1. location : Beluga_syntax.Location.t;
    2. theorems : thm_decl Support.List1.t;
    }
    (*

    Mutually recursive theorem declaration(s)

    *)
  13. | Val of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Name.t;
    3. typ : Comp.typ option;
    4. expression : Comp.exp;
    }
    (*

    Computation-level value declaration

    *)
  14. | Query of {
    1. location : Beluga_syntax.Location.t;
    2. name : Name.t option;
    3. mctx : LF.mctx;
    4. typ : LF.typ;
    5. expected_solutions : int option;
    6. maximum_tries : int option;
    }
    (*

    Logic programming query on LF type

    *)
  15. | MQuery of {
    1. location : Beluga_syntax.Location.t;
    2. typ : Comp.typ;
    3. expected_solutions : int option;
    4. search_tries : int option;
    5. search_depth : int option;
    }
    (*

    Logic programming mquery on Comp. type

    *)
  16. | Module of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : string;
    3. declarations : decl list;
    }
    (*

    Namespace declaration for other declarations

    *)
  17. | Comment of {
    1. location : Beluga_syntax.Location.t;
    2. content : string;
    }
    (*

    Documentation comment

    *)
