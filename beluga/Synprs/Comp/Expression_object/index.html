Module Comp.Expression_object

type t =
  1. | Raw_identifier of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Beluga_syntax.Identifier.t;
    3. quoted : Stdlib.Bool.t;
    }
  2. | Raw_qualified_identifier of {
    1. location : Beluga_syntax.Location.t;
    2. identifier : Beluga_syntax.Qualified_identifier.t;
    3. quoted : Stdlib.Bool.t;
    }
  3. | Raw_fn of {
    1. location : Beluga_syntax.Location.t;
    2. parameters : Beluga_syntax.Identifier.t Support.Option.t Support.List1.t;
    3. body : t;
    }
  4. | Raw_mlam of {
    1. location : Beluga_syntax.Location.t;
    2. parameters : (Beluga_syntax.Identifier.t Support.Option.t * [ `Plain | `Hash | `Dollar ]) Support.List1.t;
    3. body : t;
    }
  5. | Raw_fun of {
    1. location : Beluga_syntax.Location.t;
    2. branches : (Pattern_object.t Support.List1.t * t) Support.List1.t;
    }
  6. | Raw_let of {
    1. location : Beluga_syntax.Location.t;
    2. pattern : Pattern_object.t;
    3. scrutinee : t;
    4. body : t;
    }
  7. | Raw_impossible of {
    1. location : Beluga_syntax.Location.t;
    2. scrutinee : t;
    }
  8. | Raw_case of {
    1. location : Beluga_syntax.Location.t;
    2. scrutinee : t;
    3. check_coverage : Stdlib.Bool.t;
    4. branches : (Pattern_object.t * t) Support.List1.t;
    }
  9. | Raw_tuple of {
    1. location : Beluga_syntax.Location.t;
    2. elements : t Support.List2.t;
    }
  10. | Raw_hole of {
    1. location : Beluga_syntax.Location.t;
    2. label : Beluga_syntax.Identifier.t Support.Option.t;
    }
  11. | Raw_box_hole of {
    1. location : Beluga_syntax.Location.t;
    }
  12. | Raw_application of {
    1. location : Beluga_syntax.Location.t;
    2. expressions : t Support.List2.t;
    }
  13. | Raw_annotated of {
    1. location : Beluga_syntax.Location.t;
    2. expression : t;
    3. typ : Sort_object.t;
    }
