Module LF.Term

External LF terms.

type t =
  1. | Variable of {
    1. location : Common.Location.t;
    2. identifier : Common.Identifier.t;
    }
    (*

    Variable { identifier = "x"; _ } is the term-level variable with name "x", which is either a bound variable having a lambda binder, or a free variable having no such corresponding binder.

    *)
  2. | Constant of {
    1. location : Common.Location.t;
    2. identifier : Common.Qualified_identifier.t;
    3. operator : Common.Operator.t;
    4. quoted : Stdlib.Bool.t;
    }
    (*

    Constant { identifier = "c"; _ } is the term-level constant with qualified identifier "c".

    *)
  3. | Application of {
    1. location : Common.Location.t;
    2. applicand : t;
    3. arguments : t Support.List1.t;
    }
    (*

    Application { applicand; arguments; _ } is the term-level application of applicand with arguments.

    • If applicand = Term.Constant { operator; _ } and Operator.is_infix operator, then List.length arguments = 2.
    • If applicand = Term.Constant { operator; _ } and Operator.is_postfix operator, then List.length arguments = 1.
    *)
  4. | Abstraction of {
    1. location : Common.Location.t;
    2. parameter_identifier : Common.Identifier.t Support.Option.t;
    3. parameter_type : Typ.t Support.Option.t;
    4. body : t;
    }
    (*
    • Abstraction { parameter_identifier = Option.Some "x"; body; _ } is the term \x. body. The variable "x" ranges over LF terms.
    • Abstraction { parameter_identifier = Option.None; body; _ } is the term \_. body.
    *)
  5. | Wildcard of {
    1. location : Common.Location.t;
    }
    (*

    Wildcard { _ } is the omission _ of a fresh term-level variable.

    *)
  6. | TypeAnnotated of {
    1. location : Common.Location.t;
    2. term : t;
    3. typ : Typ.t;
    }
    (*

    TypeAnnotated { term = u; typ = t; _ } is the term u : t.

    *)
