Module Comp.Pattern

External computation-level patterns.

type t =
  1. | Variable of {
    1. location : Common.Location.t;
    2. identifier : Common.Identifier.t;
    }
  2. | Constant of {
    1. location : Common.Location.t;
    2. identifier : Common.Qualified_identifier.t;
    3. operator : Common.Operator.t;
    4. quoted : Stdlib.Bool.t;
    }
  3. | Tuple of {
    1. location : Common.Location.t;
    2. elements : t Support.List2.t;
    }
  4. | Application of {
    1. location : Common.Location.t;
    2. applicand : t;
    3. arguments : t Support.List1.t;
    }
  5. | TypeAnnotated of {
    1. location : Common.Location.t;
    2. pattern : t;
    3. typ : Typ.t;
    }
  6. | Wildcard of {
    1. location : Common.Location.t;
    }
