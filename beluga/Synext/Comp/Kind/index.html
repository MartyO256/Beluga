Module Comp.Kind

External computation-level kinds.

type t =
  1. | Ctype of {
    1. location : Common.Location.t;
    }
    (*

    Ctype { _ } is the kind of computational types ctype.

    *)
  2. | Arrow of {
    1. location : Common.Location.t;
    2. domain : Typ.t;
    3. range : t;
    }
    (*

    Arrow { domain; range; _ } is the computational kind domain -> range.

    *)
