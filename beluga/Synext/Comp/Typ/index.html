Module Comp.Typ

External computation-level types.

type t =
  1. | Constant of {
    1. location : Common.Location.t;
    2. identifier : Common.Qualified_identifier.t;
    3. operator : Common.Operator.t;
    4. quoted : Stdlib.Bool.t;
    5. variant : [ `Inductive | `Stratified | `Coinductive | `Abbreviation ];
    }
    (*

    Constant { identifier = "c"; } is the computation-level type constant "c".

    • If variant = `Inductive, then this constant was defined in an inductive type declaration.
    • If variant = `Stratified, then this constant was defined in a stratified type declaration.
    • If variant = `Coinductive, then this constant was defined in a coinductive type declaration.
    • If variant = `Abbreviation, then this constant was defined in a computation type abbreviation declaration.
    *)
  2. | Arrow of {
    1. location : Common.Location.t;
    2. domain : t;
    3. range : t;
    4. orientation : [ `Forward | `Backward ];
    }
    (*

    Arrow { domain; range; _ } is the computational type domain -> range.

    *)
  3. | Cross of {
    1. location : Common.Location.t;
    2. types : t Support.List2.t;
    }
    (*

    Cross { typs = [t1; t2; ...; tn]; _ } is the type of tuple t1 * t2 * ... * tn.

    *)
