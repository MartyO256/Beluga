Module Term.Pattern

External contextual LF term patterns.

type t =
  1. | Variable of {
    1. location : Common.Location.t;
    2. identifier : Common.Identifier.t;
    }
    (*

    Variable { identifier = "x"; _ } is the term-level variable pattern "x".

    *)
  2. | Parameter_variable of {
    1. location : Common.Location.t;
    2. identifier : Common.Identifier.t;
    }
    (*

    Parameter_variable { identifier = "#x"; _ } is the term-level parameter variable pattern with name "#x".

    *)
  3. | Substitution_variable of {
    1. location : Common.Location.t;
    2. identifier : Common.Identifier.t;
    }
    (*

    Substitution_variable { identifier = "$x"; _ } is the term-level substitution variable pattern with name "$x".

    *)
  4. | Constant of {
    1. location : Common.Location.t;
    2. identifier : Common.Qualified_identifier.t;
    3. operator : Common.Operator.t;
    4. quoted : Stdlib.Bool.t;
    }
    (*

    Constant { identifier = "c"; _ } is the term-level constant pattern "c".

    *)
  5. | Wildcard of {
    1. location : Common.Location.t;
    }
    (*

    Wildcard _ is the term-level catch-all pattern _.

    *)
  6. | Tuple of {
    1. location : Common.Location.t;
    2. terms : t Support.List1.t;
    }
    (*

    Tuple { terms; _ } is the tuple pattern <p1; p2; ...; pn> if terms = [p1; p2; ...; pn].

    *)
  7. | Projection of {
    1. location : Common.Location.t;
    2. term : t;
    3. projection : [ `By_identifier of Common.Identifier.t | `By_position of Support.Int.t ];
    }
    (*

    Projection { term; _ } is the pattern for the projection of a tuple term. This projection is used to indicate which element of a tuple the scrutinee is matched against in a case-expression.

    *)
  8. | Abstraction of {
    1. location : Common.Location.t;
    2. parameter_identifier : Common.Identifier.t Support.Option.t;
    3. parameter_type : Typ.t Support.Option.t;
    4. body : t;
    }
    (*

    Abstraction { parameter_identifier = Option.Some "x"; parameter_type = Option.Some t; body; _ } is the pattern \x:t. body.

    *)
  9. | Substitution of {
    1. location : Common.Location.t;
    2. term : t;
    3. substitution : Substitution.t;
    }
    (*

    Substitution { term; substitution; _ } is the pattern term[substitution].

    *)
  10. | Application of {
    1. location : Common.Location.t;
    2. applicand : t;
    3. arguments : t Support.List1.t;
    }
    (*

    Application { applicand; arguments; _ } is the term-level application pattern of applicand with arguments.

    • If applicand = Term.Constant { operator; _ } and Operator.is_infix operator, then List.length arguments = 2.
    • If applicand = Term.Constant { operator; _ } and Operator.is_postfix operator, then List.length arguments = 1.
    *)
  11. | TypeAnnotated of {
    1. location : Common.Location.t;
    2. term : t;
    3. typ : Typ.t;
    }
    (*

    TypeAnnotated { term = x; typ = t; _ } is the pattern x : t.

    *)
